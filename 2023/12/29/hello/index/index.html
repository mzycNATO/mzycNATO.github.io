#include<bits/stdc++.h>
#define ll long long
#define MOD 1000000007ll
using namespace std;
string L;
ll dp[100005][2],n;
ll qpow(ll a,ll b)
{
    ll res=1;
    while(b)
    {
        if(b&1)res=res*a%MOD;
        a=a*a%MOD;b>>=1;
    }
    return res;
}
ll dfs(ll now,ll fa,bool dm)
{
    if(dp[now][dm])return fa*dp[now][dm]%MOD;
    if(now==n)return fa;
    ll res=0,maxn=(dm?(L[now]-'0'):1);
    if(dm)
    {
        if(maxn==1)
        {
        res=(res+dfs(now+1,1,0))%MOD;	
        res=(res+dfs(now+1,2,1))%MOD;	
        }
        else
        res=(res+dfs(now+1,1,1))%MOD;
    }
    else
    res=qpow(3,n-now);
    dp[now][dm]=res;
    return res*fa%MOD;
}
int main()
{
    ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    cin>>L;
    n=L.size();
    ll res=0;
    res=(res+dfs(1,1,0))%MOD;
    res=(res+dfs(1,2,1))%MOD;
    cout<<res;
}
